# 테이블 복사
```SQL
CREATE TABLE DEPT_TEMP
  AS SELECT * FROM DEPT;
```

# 테이블 삭제
```SQL
DROP TABLE 테이블이름;
```


# 테이블에 데이터 추가 [INSERT INTO]
```sql
--열을 명시해하고 추가하는 방식
INSERT INTO DEPT_TEMP(DEPTNO, DNAME, LOC)
            VALUES(50,'DATABASE','SEOUL');

SELECT * FROM DEPT_TEMP;

--열 순서, 형태 등을 맞추어 열명시하지 않고 추가하기
INSERT INTO DEPT_TEMP
    VALUES(60, 'NETWORK','BUSAN');

SELECT * FROM DEPT_TEMP;

--NULL 추가하기
INSERT INTO DEPT_TEMP (DEPTNO, DNAME, LOC)
    VALUES(70,'WEB',NULL)   -- > 자료형이 문자열 또는 날짜형일 경우 빈칸 '' 으로 해 주어도 같다. 하지만 NULL로 명시하는게 좋다.
    
SELECT * FROM DEPT_TEMP;


--열을 제외하고 넣으면 제외 된 것에 자동으로 NULL추가
INSERT INTO DEPT_TEMP (DEPTNO, LOC)
    VALUES(90, 'INCHEON');
SELECT * FROM DEPT_TEMP;

```

## 날짜 데이터 입력
```SQL
-- 일반적
INSERT INTO EMP_TEMP (EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO)
    VALUES(9999,'홍길동','PRESIDENT',NULL,'2001/01/01',5000,1000,10);


-- / 를 이용
INSERT INTO EMP_TEMP (EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO)
    VALUES(1111,'성춘향','MANAGER',9999,'2001/01/05',4000,NULL,20);
  
-- TO_DATE 이용
INSERT INTO EMP_TEMP(EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO)
    VALUES(2111,'이순신','MANAGER',9999,TO_DATE('07/01/2001','DD/MM/YYYY'),4000,NULL,20)
    
-- SYSDATE 이용

INSERT INTO EMP_TEMP (EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO)
    VALUES(3111,'심청이','MANAGER',9999,SYSDATE,4000,NULL,30);
```

## 서브쿼리 사용 한번에 여러 데이터 추가 ★
```SQL
INSERT INTO EMP_TEMP(EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO)
    SELECT E.EMPNO, E.ENAME, E.JOB, E.MGR, E.HIREDATE, E.SAL, E.COMM, E.DEPTNO
        FROM EMP E, SALGRADE S
       WHERE E.SAL BETWEEN LOSAL AND HISAL
       AND S.GRADE=1;
```

# 테이블의 데이터 수정 [UPDATE + SET + (WHERE)]
```SQL
UPDATE 변경할 테이블명
SET col1=내용, col2=내용,...
WHERE 행조건
```

## 데이터 전체 수정하기
```sql
UPDATE DEPT_TEMP2
    SET LOC = 'SEOUL';
```

## 데이터 일부분만 수정하기
```sql
UPDATE DEPT_TEMP2
    SET DNAME = 'DATABASE',
        LOC = 'SEOUL'
  WHERE DEPTNO=40;
```

## 서브쿼리를 사용하여 데이터 수정하기
```SQL
UPDATE DEPT_TEMP2
    SET LOC =  'SEOUL'
  WHERE DEPTNO = (SELECT DEPTNO
                    FROM DEPT_TEMP2
                   WHERE DNAME='OPERATIONS');
```
